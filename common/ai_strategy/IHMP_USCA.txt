	reversed = yes

	enable = {
		any_country = {
			is_faction_leader = yes
			OR = {
				tag = USA
				is_in_faction_with = USA
			}
		}
		has_game_rule = {
			rule = MEX_ai_behavior
			option = MEX_MP_1
		}
		USA = { has_defensive_war = yes }
		MEX = { has_completed_focus = MEX_focus_liberty_and_justice_for_all }
	}

	abort_when_not_enabled = yes

	ai_strategy = {
		type = alliance
		id = MEX
		value = 200
	}
}
