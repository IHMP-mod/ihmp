FRA_MP_Arty_Production = {

	allowed = {
		tag = FRA
	}

	enable = {
		OR = {
			has_game_rule = {
				rule = FRA_ai_behavior
				option = FRA_MP_1
			}
			has_game_rule = {
				rule = FRA_ai_behavior
				option = FRA_MP_2
			}
		}
		has_capitulated = no
		has_focus_tree = french_focus
		OR = {
			NOT = {
				has_equipment = {
					artillery_equipment_1 > 0
				}
			}
			NOT = {
				has_equipment = {
					artillery_equipment_2 > 0
				}
			}
			NOT = {
				has_equipment = {
					artillery_equipment_3 > 0
				}
			}
		}
	}

	abort = {
		OR = {
			has_capitulated = yes
			OR = {
				has_equipment = {
					artillery_equipment_1 > 100
				}
				has_equipment = {
					artillery_equipment_2 > 100
				}
				has_equipment = {
					artillery_equipment_3 > 100
				}
			}
		}
		
	}
	ai_strategy = {
		type = equipment_production_min_factories
		id = artillery
		value = 2
	}
}
